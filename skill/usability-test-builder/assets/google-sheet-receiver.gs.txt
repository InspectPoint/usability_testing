// ── Inspect Point usability-test → Google Sheet receiver ─────────────────────
// Paste this whole file into the Apps Script editor attached to your sheet,
// then deploy it as a Web App (see USABILITY-TEST-README / Claude's steps).
// It writes one row per participant and creates the header row automatically.

function doPost(e) {
  var lock = LockService.getScriptLock();
  lock.waitLock(30000);
  try {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0];
    var data = JSON.parse(e.postData.contents);

    var row = {
      'Recorded at': data.finishedAt || new Date().toISOString(),
      'Test': data.test || '',           // labels the row so the dashboard can group by project
      'Participant': data.participant || '',
      'Started at': data.startedAt || '',
      'Finished at': data.finishedAt || ''
    };

    var tasks = data.tasks || [];
    for (var i = 0; i < tasks.length; i++) {
      var t = tasks[i], n = i + 1;
      row['Task ' + n + ' name'] = t.id || '';
      row['Task ' + n + ' completed'] = t.completed ? 'Yes' : 'No';
      row['Task ' + n + ' seconds'] = t.seconds || '';
      row['Task ' + n + ' clicks'] = t.clicks || '';
      row['Task ' + n + ' click path'] = (t.path || []).join(' > ');
    }

    // Survey questions vary per test, so store each answer generically as "Q: <name>".
    // The dashboard treats numeric answers as ratings (averaged) and text as open responses.
    var s = data.survey || {};
    Object.keys(s).forEach(function (k) { row['Q: ' + k] = (s[k] === undefined ? '' : s[k]); });
    row['Device'] = data.userAgent || '';

    var headers = Object.keys(row);
    if (sheet.getLastRow() === 0) {
      sheet.appendRow(headers);
    } else {
      var existing = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
      var missing = headers.filter(function (h) { return existing.indexOf(h) === -1; });
      if (missing.length) sheet.getRange(1, existing.length + 1, 1, missing.length).setValues([missing]);
    }

    var current = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
    var out = current.map(function (h) { return row[h] !== undefined ? row[h] : ''; });
    sheet.appendRow(out);

    return ContentService.createTextOutput('ok');
  } finally {
    lock.releaseLock();
  }
}
